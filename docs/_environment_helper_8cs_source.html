43  internal static bool Is64BitOperatingSystem()
44  {
45  // Check if this process is natively an x64 process. If it is, it will only run on x64 environments, thus, the environment must be x64.
46  if (IntPtr.Size == 8)
47  return true;
48 
49  // Check if this process is an x86 process running on an x64 environment.
50  var moduleHandle = NativeMethods.GetModuleHandle("kernel32");
51  if (moduleHandle == IntPtr.Zero)
52  return false;
53 
54  var processAddress = NativeMethods.GetProcAddress(moduleHandle, "IsWow64Process");
55  if (processAddress == IntPtr.Zero)
56  return false;
57 
58  bool result;
59  return NativeMethods.IsWow64Process(NativeMethods.GetCurrentProcess(), out result) && result;
60  }
