137  if (EnvironmentHelper.Is64BitProcess() && EnvironmentHelper.Is64BitOperatingSystem())
138  {
139  // We are running 64-bit!

141  }
142  else
143  {
144  // We are running 32-bit!
145  chromaSdkPointer = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK.dll");
146  }
147

436  private static T GetDelegateFromLibrary<T>(IntPtr lib, string name)
437  {
438  var functionPtr = Native.Kernel32.NativeMethods.GetProcAddress(lib, name);
439  if (functionPtr == IntPtr.Zero)
440  {
441  throw new ColoreException(
442  "Failed to dynamically load function, GetProcAddress returned NULL for " + name);
443  }
444 
445  return (T)(object)Marshal.GetDelegateForFunctionPointer(functionPtr, typeof(T));
446  }
