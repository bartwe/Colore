288  Justification = "Parameter names match those in the Message struct.")]
289  public bool HandleMessage(IntPtr handle, int msgId, IntPtr wParam, IntPtr lParam)
290  {
291  if (!_registered)
292  throw new InvalidOperationException("Register must be called before event handling can be performed.");
293 
294  if (handle != _registeredHandle)
295  {
296  throw new ArgumentException(
297  "The specified window handle does not match the currently registered one.",
298  nameof(handle));
299  }
300 
301  if (msgId != Constants.WmChromaEvent)
302  return false;
303 
304  var handled = false;
305 
306  var type = wParam.ToInt32();
307  var state = lParam.ToInt32();
308 
309  switch (type)
310  {
311  case 1: // Chroma SDK support
312  OnSdkSupport(state != 0);
313  handled = true;
314  break;
315 
316  case 2: // Access to device
317  OnDeviceAccess(state != 0);
318  handled = true;
319  break;
320 
321  case 3: // Application state
322  OnApplicationState(state != 0);
323  handled = true;
324  break;
325  }
326 
327  return handled;
328  }
