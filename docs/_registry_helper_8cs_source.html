54  internal static bool IsSdkAvailable()
55  {
56  bool dllValid;
57 
58 #if ANYCPU
59  if (EnvironmentHelper.Is64BitProcess() && EnvironmentHelper.Is64BitOperatingSystem())
60  dllValid = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK64.dll") != IntPtr.Zero;
61  else
62  dllValid = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK.dll") != IntPtr.Zero;
63 #elif WIN64
64  dllValid = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK64.dll") != IntPtr.Zero;
65 #else
66  dllValid = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK.dll") != IntPtr.Zero;
67 #endif
68 
69  bool regEnabled;
70 
71  try
72  {
73  using (var key = Registry.LocalMachine.OpenSubKey(SdkRegKeyPath))
74  {
75  if (key != null)
76  {
77  var value = key.GetValue("Enable");
78 
79  if (value is int)
80  regEnabled = (int)value == 1;
81  else
82  {
83  regEnabled = true;
84  Log.Warn(
85  "Chroma SDK has changed registry setting format. Please update Colore to latest version.");
86  Log.DebugFormat("New Enabled type: {0}", value.GetType());
87  }
88  }
89  else
90  regEnabled = false;
91  }
92  }
93  catch (SecurityException ex)
94  {
95  // If we can't access the registry, best to just assume
96  // it is enabled.
97  Log.Warn("System raised SecurityException during read of SDK enable flag in registry.", ex);
98  regEnabled = true;
99  }
100  catch (UnauthorizedAccessException ex)
101  {
102  // If we can't access the registry, best to just assume
103  // it is enabled.
104  Log.Warn("Not authorized to read registry for SDK enable flag.", ex);
105  regEnabled = true;
106  }
107 
108  return dllValid && regEnabled;
109  }

119  internal static bool TryGetSdkVersion(out SdkVersion ver)
120  {
121  try
122  {
123  using (var key = Registry.LocalMachine.OpenSubKey(SdkRegKeyPath, false))
124  {
125  if (key == null)
126  {
127  Log.Warn("Failed to open registry key to read SDK version.");
128  ver = new SdkVersion(0, 0, 0);
129  return false;
130  }
131 
132  var major = key.GetValue("MajorVersion");
133  var minor = key.GetValue("MinorVersion");
134  var revision = key.GetValue("RevisionNumber");
135 
136  if (major is int && minor is int && revision is int)
137  {
138  ver = new SdkVersion((int)major, (int)minor, (int)revision);
139  return true;
140  }
141 
142  Log.WarnFormat(
143  "Unknown version component types, please tell a developer: Ma is {0}, Mi is {1}, R is {2}.",
144  major.GetType(),
145  minor.GetType(),
146  revision.GetType());
147 
148  ver = new SdkVersion(0, 0, 0);
149  return false;
150  }
151  }
152  catch (SecurityException ex)
153  {
154  Log.Warn("System raised SecurityException during read of SDK version in registry.", ex);
155  ver = new SdkVersion(0, 0, 0);
156  return false;
157  }
158  catch (UnauthorizedAccessException ex)
159  {
160  Log.Warn("Not authorized to read registry for SDK version.", ex);
161  ver = new SdkVersion(0, 0, 0);
162  return false;
163  }
164  }
