143  private static TraceSource GetParent(string name, SourceLevels level)
144  {
145  if (string.IsNullOrEmpty(name))
146  return new TraceSource("Default", level);
147 
148  var source = new TraceSource(name, level);
149 
150  if (IsConfigured(source))
151  return source;
152 
153  var dotIndex = name.LastIndexOf(".", StringComparison.Ordinal);
154 
155  // ReSharper disable once TailRecursiveCall
156  return dotIndex == -1 ? GetParent(null, level) : GetParent(name.Substring(0, dotIndex), level);
157  }

162  private void Configure()
163  {

175
