212  internal static DeviceInfo QueryDevice(Guid id)
213  {
214  var ptr = Marshal.AllocHGlobal(Marshal.SizeOf(typeof(DeviceInfo)));
215 
216  try
217  {
218  var result = NativeMethods.QueryDevice(id, ptr);
219 
220  if (!result)
221  throw new NativeCallException("QueryDevice", result);
222 
223  if (ptr == IntPtr.Zero)
224  throw new ColoreException("Device query failed, ptr NULL.");
225 
226  var info = (DeviceInfo)Marshal.PtrToStructure(ptr, typeof(DeviceInfo));
227 
228  return info;
229  }
230  finally
231  {
232  Marshal.FreeHGlobal(ptr);
233  }
234  }
